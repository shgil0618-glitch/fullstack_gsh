테이블
<br>
```
CREATE TABLE INGREDIENT (
    ingredientid    NUMBER(6)        CONSTRAINT pk_ingredient PRIMARY KEY,   -- 재료 고유번호
    name             VARCHAR2(100)    CONSTRAINT nn_ingredient_name NOT NULL, -- 재료명
    type             VARCHAR2(50),                                         -- 종류 (채소, 육류, 해산물, 소스 등)
    kcalper100g    NUMBER(5),                                            -- 100g당 칼로리
    allergyflag     VARCHAR2(100),                                       -- 알러지 정보 (쉼표로 구분 가능)
    createdat       DATE DEFAULT SYSDATE,                                -- 등록일
    updatedat       DATE DEFAULT SYSDATE                               -- 수정일
);
```
create sequence INGREDIENT_seq;

desc INGREDIENT;
